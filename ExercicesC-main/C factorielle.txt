#include <stdio.h>

int somme(int n)
{
    int somme = 0;
    for (int i = 1; i <= n; i++)
    {
        somme += i;
    }
    return somme;
}

int factorielle(int n)
{
    int factorielle = 1;
    for (int i = 1; i <= n; i++)
    {
        somme *= i;
    }
    return factorielle;
}

int main ()
{

    int n;
    scanf ("%d", & n ) ;
    printf ("%d\n", somme(n) );

}
