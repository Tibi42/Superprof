#include <stdio.h>

int maximum(int tableau[]) {
	int a = tableau[0] ;
    int b;
    int i;
    for(i=1; i<10; i++){
    	b = tableau[i];
    	if(a < b){
    		a = b;
    	}
    }
    return a;
}

int main() {
  int myNumbers[] = {1, 50, 75, 100, 5, 1, 75, 3, 100, 1};
	
  int m;
  m = maximum(myNumbers);
  printf("Le maximum du tableau vaut %d.\n", m);
  return 0;
}
